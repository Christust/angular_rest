# Iniciar el proyecto
```
ng new angular_rest
```
```
//app.module.ts
...
imports: [
    BrowserModule,
    AppRoutingModule,
    BrowserAnimationsModule,
    // Importamos FormsModule para utilizar ngModel y bindear variables
    FormsModule,
  ],
...
```

# Generar componentes
```
ng g c pages/HomePage
...
ng g c pages/XPage
```
```
//app.routing.module.ts
const routes: Routes = [
  {
    path: "",
    pathMatch: "full",
    redirectTo: "home",
  },
  {
    path: "home",
    component: HomePageComponent,
  },
  ...,
  {
    path: "**",
    component: NotFoundPageComponent,
  }
];
```
